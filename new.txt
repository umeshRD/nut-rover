  <script>
    const CAMERA_IP = "192.168.4.1";  // Change to your ESP32 camera IP
    const SENSOR_IP = "192.168.4.2";  // Change to your ESP32 sensor IP

    setInterval(() => {
      fetch(`http://${SENSOR_IP}/data`)
        .then(res => res.json())
        .then(data => {
          document.getElementById("temperature").innerText = `${data.temp.toFixed(1)}°C`;
          document.getElementById("humidity").innerText = `${data.hum.toFixed(1)}%`;
          document.getElementById("speed").innerText = `${data.speed.toFixed(1)} km/h`;
          document.getElementById("rpm").innerText = `${data.rpm}`;
        }).catch(() => {
          document.getElementById("temperature").innerText = '--°C';
          document.getElementById("humidity").innerText = '--%';
          document.getElementById("speed").innerText = '-- km/h';
          document.getElementById("rpm").innerText = '--';
        });
    }, 2000);

    function setCamera(setting, value) {
      fetch(`http://${CAMERA_IP}/control?var=${setting}&val=${value}`);
    }

    function toggleFlash() {
      fetch(`http://${CAMERA_IP}/led?var=led_intensity&val=255`);
    }

    function flipVertical() {
      const video = document.getElementById("video");
      video.style.transform = video.style.transform.includes("scaleY(-1)")
        ? video.style.transform.replace("scaleY(-1)", "scaleY(1)")
        : video.style.transform.replace("scaleY(1)", "scaleY(-1)");
    }

    function flipHorizontal() {
      const video = document.getElementById("video");
      if (video.style.transform.includes("scaleX(-1)")) {
        video.style.transform = video.style.transform.replace("scaleX(-1)", "scaleX(1)");
      } else if (video.style.transform.includes("scaleX(1)")) {
        video.style.transform = video.style.transform.replace("scaleX(1)", "scaleX(-1)");
      } else {
        video.style.transform += " scaleX(-1)";
      }
    }
  </script>
