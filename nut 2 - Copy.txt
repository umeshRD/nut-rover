<script>
  const ESP32_IP = "192.168.8.157";

  function setCamera(setting, value) {
    fetch(`http://${ESP32_IP}/control?var=${setting}&val=${value}`)
      .then(() => updateStatus(`Camera setting '${setting}' set to ${value}`))
      .catch(() => updateStatus("Camera control error"));
  }

  function moveRover(direction) {
    fetch(`http://${ESP32_IP}/move?dir=${direction}`)
      .then(() => updateStatus(`Rover moving: ${direction}`))
      .catch(() => updateStatus("Failed to send movement command"));
  }

  function updateStatus(message) {
    const status = document.getElementById("status");
    status.innerText = message;
    setTimeout(() => { status.innerText = ""; }, 3000);
  }

  function changeResolution(val) {
    fetch(`http://${ESP32_IP}/control?var=framesize&val=${val}`)
      .then(() => updateStatus("Resolution changed"))
      .catch(() => updateStatus("Failed to change resolution"));
  }

  setInterval(() => {
    fetch(`http://${ESP32_IP}/data`)
      .then(res => res.json())
      .then(data => {
        document.getElementById("temp").innerText = data.temp.toFixed(1) + " °C";
        document.getElementById("hum").innerText = data.hum.toFixed(1) + " %";
        document.getElementById("speed").innerText = data.speed.toFixed(1) + " km/h";
        document.getElementById("rpm").innerText = data.rpm;
      })
      .catch(() => {
        document.getElementById("temp").innerText = "-- °C";
        document.getElementById("hum").innerText = "-- %";
        document.getElementById("speed").innerText = "-- km/h";
        document.getElementById("rpm").innerText = "--";
      });
  }, 2000);
</script>
